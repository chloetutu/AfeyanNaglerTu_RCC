4 Create larger cell type groups according to Alex:

Notes on contamination: B: “B.MEM”, “B.PLASMA” alex found some CD3D+ subclusters Myeloid: “cDC”, “MONO.NONCLASS”, “MONO.CLASS”, “M2.MAC”, “MAC”, “MAST”, “pDC” alex found some CD3D+ CD8+ subclusters T: “CD8.TEX”,“CD4.TMEM”, “CD8.TRM”, “TAP”, “TPROL”, “CD4.TREG”, “CD8.IFN” alex found some myeloid contaminant in Tprolif NK: “NK1”, “NK2” RCC: “RCC1”, “RCC2”, “RCC3” i found some CD68 and PTPRC expression Endo: “ENDO”

scrna@meta.data <- scrna@meta.data %>%
  mutate(generalized_clusters = case_when(unified_cluster %in% c("B.MEM", "B.PLASMA") ~ "B",
                                          unified_cluster %in% c("cDC", "MONO.NONCLASS", "MONO.CLASS", "M2.MAC", "MAC", "MAST", "pDC") ~ "Myeloid",
                                          unified_cluster %in% c("CD8.TEX","CD4.TMEM", "CD8.TRM", "TAP", "TPROL", "CD4.TREG", "CD8.IFN") ~ "T",
                                          unified_cluster %in% c("NK1", "NK2") ~ "NK",
                                          unified_cluster %in% c("RCC1", "RCC2", "RCC3") ~ "RCC",
                                          T ~ unified_cluster))
