12 Cast TLS to a singular value

format_md <- function(obj){
  obj <- obj %>%
    mutate(final_area = case_when(area %in% c("TLS1", "TLS2", "TLS3", "TLS4", "M2") ~ "TLS",
                                T ~ area),
         Patient = str_sub(orig.ident, start = 1, end = 5),
         tls_bool = case_when(final_area == "TLS" ~ "TLS",
                              T ~ "nonTLS"),
         boundary_bool = case_when(final_area == "Boundary" ~ "Boundary",
                                   T ~ "nonBoundary"),
         tumor_bool = case_when(final_area == "Tumor" ~ "Tumor",
                                   T ~ "nonTumor"),
         Patient_tls = paste0(Patient, "_", tls_bool),
         Patient_boundary = paste0(Patient, "_", boundary_bool),
         Patient_tumor = paste0(Patient, "_", tumor_bool),
         dummy = "dummy")
  
  return(obj)
}
