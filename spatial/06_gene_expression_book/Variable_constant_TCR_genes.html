compare_tra_constant_variable <- function(obj, puck_name){
  constant <- Cells(subset(x = obj, subset = TRAC > 1))

  variable <- obj@meta.data %>%
    filter(str_detect(chains, "TRA")) %>%
    pull(barcode)

  plot(euler(list(Constant_TRAC = constant, Variable = variable)),
       main = puck_name, 
       quantities = TRUE)
}

10 Fraction of beads with variable TCRs functions

add_trac <- function(obj){
  trc_gene_exp <- FetchData(object = obj, vars = c("TRAC"), layer = "counts")
  obj <- AddMetaData(obj, trc_gene_exp)
  return(obj@meta.data)
}

calculate_pct_tra_beads_with_tcr <- function(obj){
  md <- add_trac(obj)
  
  tra_df <- data.frame(matrix(nrow = 0, ncol = 6, dimnames = list(rownames = NULL, colnames = c("orig.ident", "n_total_beads", "n_umi", "n_tra_beads", "n_tra_with_tcr_beads", "pct_tra_with_tcr"))))

  expr <- unique(md$TRAC)
  for (i in c(expr)){
    puck <- unique(md$orig.ident)
    num_total_beads <- md %>% nrow()
    # Number of beads with N TRAC counts
    num_beads_tra <- md %>% filter(TRAC == i) %>% nrow()
    # Number of beads with N TRAC counts and a variable TCRA
    num_beads_tra_tcr <- md %>% filter(TRAC == i, str_detect(chains, "TRA")) %>% nrow()
    
    pct_tra_with_tcr <- (num_beads_tra_tcr/num_beads_tra)*100
    
    dat <- data.frame(orig.ident = puck, n_total_beads = num_total_beads, n_umi = i, n_tra_beads = num_beads_tra, n_tra_with_tcr_beads = num_beads_tra_tcr, pct_tra_with_tcr = pct_tra_with_tcr)
    
    tra_df <- rbind(tra_df, dat)
  }
  
  return(tra_df)
}

add_trbc <- function(obj){
  trc_gene_exp <- FetchData(object = obj, vars = c("TRBC2", "TRBC1"), layer = "counts")
  obj <- AddMetaData(obj, trc_gene_exp)
  
  return(obj@meta.data)
}

calculate_pct_trb_beads_with_tcr <- function(obj){
  md <- add_trbc(obj)
  
  md <- md %>%
    rowwise() %>%
    mutate(TRBC = sum(TRBC1, TRBC2)) %>%
    ungroup()
  
  trb_df <- data.frame(matrix(nrow = 0, ncol = 6, dimnames = list(rownames = NULL, colnames = c("orig.ident", "n_total_beads", "n_umi", "n_trb_beads", "n_trb_with_tcr_beads", "pct_trb_with_tcr"))))

  expr <- unique(md$TRBC)
  for (i in c(expr)){
    puck <- unique(md$orig.ident)
    num_total_beads <- md %>% nrow()
    num_beads_trb <- md %>% filter(TRBC == i) %>% nrow()
    num_beads_trb_tcr <- md %>% filter(TRBC == i, !is.na(all_CDR3)) %>% nrow()
    
    pct_trb_with_tcr <- (num_beads_trb_tcr/num_beads_trb)*100
    
    dat <- data.frame(orig.ident = puck, n_total_beads = num_total_beads, n_umi = i, n_trb_beads = num_beads_trb, n_trb_with_tcr_beads = num_beads_trb_tcr, pct_trb_with_tcr = pct_trb_with_tcr)
    
    trb_df <- rbind(trb_df, dat)
  }
  
  return(trb_df)
}


add_trc_exp <- function(obj){
  trc_gene_exp <- FetchData(object = obj, vars = c("TRAC", "TRBC1", "TRBC2"), layer = "counts")
  obj <- AddMetaData(obj, trc_gene_exp)
  obj@meta.data <- obj@meta.data %>%
    rowwise() %>%
    mutate(TRBC = sum(TRBC1, TRBC2))
  
  return(obj)
}
