add_TLS_column <- function(obj){
  obj <- obj %>%
    mutate(tls_status = case_when(area %in% c("TLS1", "TLS2", "TLS3", "TLS4", "M2") ~ "inTLS",
                              area %in% c("Boundary", "Tumor") ~ "outTLS"))
  
  return(obj)
}

  md <- md %>%
    mutate(shared_tex_id = case_when(str_detect(sctcr_clone, "p102_295($|,)") ~ "p102_295",
                                  str_detect(sctcr_clone, "p102_289($|,)") ~ "p102_289",
