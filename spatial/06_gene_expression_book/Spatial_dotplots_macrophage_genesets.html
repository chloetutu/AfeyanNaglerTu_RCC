add_signature_score <- function(puck){
  puck <- puck %>%
    AddModuleScore(features = list(ma_gs$`Angio-TAMs`), name = "Ma_Angio-TAM") %>%
    AddModuleScore(features = list(ma_gs$`Classical TIMs`), name = "Ma_Class-TIM") %>%
    AddModuleScore(features = list(ma_gs$`IFN-TAMs`), name = "Ma_IFN-TAM") %>%
    AddModuleScore(features = list(ma_gs$`Inflam-TAMs`), name = "Ma_Inflam-TAM") %>%
    AddModuleScore(features = list(ma_gs$`Intermediate monocytes`), name = "Intermed_mono") %>%
    AddModuleScore(features = list(ma_gs$`LA-TAMs`), name = "Ma_LA-TAM") %>%
    AddModuleScore(features = list(ma_gs$`Nonclassical monocytes`), name = "Ma_Nonclass_mono") %>%
    AddModuleScore(features = list(ma_gs$`Prolif-TAMs`), name = "Ma_Prolif-TAM") %>%
    AddModuleScore(features = list(ma_gs$`Reg-TAMs`), name = "Ma_Reg-TAM") %>%
    AddModuleScore(features = list(ma_gs$`RTM-TAMs_Alveolar-RTM-like`), name = "Ma_RTM_Alv")%>%
    AddModuleScore(features = list(ma_gs$`RTM-TAMs_Interstitial-RTM-like`), name = "Ma_RTM_Int") %>%
    AddModuleScore(features = list(ma_gs$`RTM-TAMs_Kupffer-like`), name = "Ma_RTM_Kupf") %>%
    AddModuleScore(features = list(ma_gs$`RTM-TAMs_Microglia-like`), name = "Ma_RTM_Micro") %>%
    AddModuleScore(features = list(ma_gs$`RTM-TAMs_MT-RTM-like`), name = "Ma_RTM_MT")
  return(puck)
}

format_md <- function(obj){
  obj <- obj %>%
    mutate(final_area = case_when(area %in% c("TLS1", "TLS2", "TLS3", "TLS4", "M2") ~ "TLS",
                                T ~ area),
         Patient = str_sub(orig.ident, start = 1, end = 5),
         tls_bool = case_when(final_area == "TLS" ~ "TLS",
                              T ~ "nonTLS"),
         boundary_bool = case_when(final_area == "Boundary" ~ "Boundary",
                                   T ~ "nonBoundary"),
         tumor_bool = case_when(final_area == "Tumor" ~ "Tumor",
                                   T ~ "nonTumor"),
         Patient_tls = paste0(Patient, "_", tls_bool),
         Patient_boundary = paste0(Patient, "_", boundary_bool),
         Patient_tumor = paste0(Patient, "_", tumor_bool),
         dummy = "dummy")
  
  return(obj)
}
