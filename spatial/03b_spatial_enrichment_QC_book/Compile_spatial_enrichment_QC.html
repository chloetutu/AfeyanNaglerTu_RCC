13.8.1 TLS clones

TLS_clone_original <- orig_tls %>%
  select(-Region) %>%
  mutate(Analysis = "Original",
         text_size = count_in_area/(2+10),
         sig_qval = case_when(qval <= sig_level ~ "Significant",
                         qval > sig_level ~ "Non-significant")) %>%
  filter(type == "clone")

TLS_clone_remove <- TLS_clone_remove %>%
  mutate(Analysis = "remove") %>%
  filter(type == "clone")

compare_tls_clone <- TLS_clone_original %>%
  rbind(TLS_clone_remove) %>%
  mutate(phenotype = case_when(str_detect(clone,"p10") ~ "CD8TEx",
                               T ~ "Spatial"))
  
sig_clones <- compare_tls_clone %>%
  filter(qval < sig_level) %>%
  pull(clone)

sig_compare_tls_clone <- compare_tls_clone %>%
  filter(clone %in% sig_clones)

compare_tls_clone %>%
  ggplot(aes(x = enrichment, y = -log10(qval), label = ifelse(qval <= sig_level & phenotype == "CD8TEx" & Analysis == "Original", clone, NA), shape = sig_qval)) +
  geom_hline(color = "grey50", yintercept = -log10(sig_level), linetype = "longdash") +
  geom_vline(color = "black", xintercept = 0) +
  geom_hline(color = "black", yintercept = 0) +
  geom_point(aes(fill = ifelse(qval <= sig_level, phenotype, ""), color = ifelse(qval <= sig_level, Patient, ""), size = count_in_area), shape = 21, stroke = 2) +
  geom_text_repel(alpha = 1, aes(size = 500)) +
  geom_line(data = sig_compare_tls_clone, aes(group = clone), linewidth = 1, linetype = "dotted") +
  theme_classic() + 
  xlab("Enrichment in TLS") +
  ylab("-log10(adjusted pval)") +
  scale_fill_manual(values = c("white","red", "black")) +
  scale_color_manual(values = c("grey70", "#FF9300","#009193")) +

         size = guide_legend(title = "TCR Beads in Region", override.aes = aes(label = ""))) +
  coord_cartesian(ylim = c(-0.3, 3), xlim = c(-2, 6)) +
  scale_size(breaks=c(1, 31.62, 1000), limits = c(1, 1000))
