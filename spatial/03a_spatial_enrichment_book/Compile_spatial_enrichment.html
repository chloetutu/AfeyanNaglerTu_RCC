19.0.0.0.0.1 Joining the enrichment values of clones in the boundary, tumor and TLS across patients for a summarized visualization in volcano plots. Enrichment scores and p-vals (Permutation test) were calculated for clones with at least 5 beads across pucks in a patient. P-vals were FDR corrected using the BH (FDR) method.
19.0.0.0.0.2 Note: current_in_all = the number of beads with the clone (or specificity) across all areas in a patient. current_in_given_area = the number of beads with the clone across a given area in a patient.

TLS_clone <- do.call(rbind, list(P102TA_clone_tls, P107TA_clone_tls, P108TA_clone_tls)) %>%
  mutate(text_size = count_in_area/(12),
         sig_qval = case_when(qval <= sig_level ~ "Significant",
                         qval > sig_level ~ "Non-significant")) %>%
  left_join(scrna_phenotypes, by = c("clone" = "sctcr_clone"))

25.1 TLS clones

TLS_clone %>% 
  ggplot(aes(x = enrichment, y = -log10(qval), 
             label = ifelse(qval <= sig_level, clone, NA), 
             shape = sig_qval)) +
  geom_hline(color = "grey50", yintercept = -log10(sig_level), linetype = "longdash") +
  geom_vline(color = "black", xintercept = 0) +
  geom_hline(color = "black", yintercept = 0) +
  geom_point(aes(fill = ifelse(qval <= sig_level, PrimaryCluster, ""), 
                 color = ifelse(qval <= sig_level, Patient, ""), 
                 size = count_in_area), shape = 21, stroke = 2) +
  geom_text_repel(alpha = 1, aes(size = 500)) +
  theme_classic() + 
  xlab("Enrichment in TLS") +
  ylab("-log10(adjusted pval)") +
  scale_fill_manual(values = c("white","red", "black")) +
  scale_color_manual(values = c("grey70", "#FF9300","#009193")) +

  coord_cartesian(ylim = c(-0.3, 3), xlim = c(-2, 6)) +
