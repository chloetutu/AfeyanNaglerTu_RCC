4.1 P102 Puck10

# Remove empty beads
P102TA_10_RNA <- subset(P102TA_10_RNA, subset = area == "Empty", invert = TRUE)

# Remove non-assigned beads
P102TA_10_assigned <- P102TA_10_RNA@meta.data %>%
  filter(!is.na(assignment1))

p10 <- ggplot() +
  geom_point(data = arrange(P102TA_10_assigned, assignment1_ord), alpha = 1, aes(x = plot_x, y = plot_y, color = assignment1), size = 0.4) +
  celltype_color +
  coord_fixed(ratio = 1, xlim = c(0, 5000), ylim = c(0, 5000)) +
  theme_void() +
  ggtitle("P102TA 10")

p10_sep <- ggplot() +
  geom_point(data = arrange(P102TA_10_assigned, assignment1_ord), alpha = 1, aes(x = plot_x, y = plot_y, color = assignment1), size = 0.1) +
  celltype_color +
  coord_fixed(ratio = 1, xlim = c(0, 5000), ylim = c(0, 5000)) +
  theme_void() +
  ggtitle("P102TA 10") +
  facet_wrap(~assignment1)
