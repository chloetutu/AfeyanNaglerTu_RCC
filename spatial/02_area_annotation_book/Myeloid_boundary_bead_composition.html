get_pct_myeloid_in_bin <- function(obj, bin_length = 30){
  md <- obj@meta.data %>%
    select(plot_x, plot_y, area, assignment1)
  
  # Create bins for x and y
  md$x_bin <- cut(md$plot_x, breaks = bin_length, labels = FALSE) 
  md$y_bin <- cut(md$plot_y, breaks = bin_length, labels = FALSE)
  
  # Don't consider empty areas
  md <- md %>%
    filter(area != "Empty")
  bin_df <- data.frame()

  for(i in c(1:bin_length)){
    for(j in c(1:bin_length)){
      bin <- md %>%
        filter(x_bin == i,
               y_bin == j)
      boundary_count <- bin %>%
        filter(area == "Boundary") %>%
        dplyr::count() %>%
        pull(n)
      tumor_count <- bin %>%
        filter(area == "Tumor") %>%
        dplyr::count() %>%
        pull(n)
      tls_count <- bin %>%
        filter(area %in% c("TLS1", "TLS2", "TLS3", "TLS4", "M2")) %>%
        dplyr::count() %>%
        pull(n)
      total_count <- bin %>%
        dplyr::count() %>%
        pull(n)
      myeloid_count <- bin %>%
        filter(assignment1 == "Myeloid") %>%
        dplyr::count() %>%
        pull(n)
      
      bin_summary <- c(i, j, total_count, boundary_count/total_count, tumor_count/total_count, tls_count/total_count, myeloid_count, myeloid_count/total_count)
      bin_df <- rbind(bin_df, bin_summary)
    }
  }
  colnames(bin_df) <- c("x_bin", "y_bin", "total_beads", "pct_boundary", "pct_tumor", "pct_tls", "ct_myeloid", "pct_myeloid")
  
  return(bin_df)
}

assign_bin_annotation <- function(bin_df){
  bin_df <- bin_df %>%
  mutate(area = case_when(pct_tls > 0.8 ~ "TLS",
                          pct_boundary > 0.8 ~ "Boundary",
                          pct_tumor > 0.8 ~ "Tumor",
                          total_beads == 0 ~ "Empty",
                          T ~ "Mix"))
  return(bin_df)
}

31.1 Get permuted test statistic

# Get permuted test statistic for each gene set
n_perm <- 1000

set.seed(123)

n_run <- 1
n_saved <- 1
non_conv <- 0

t_val_perm.vec <- c()
# Do 1000 converging loops
while(n_saved <= n_perm){
  # Print every saved 100 perms
  if (n_saved %% 100 == 0) { print(n_saved) }
  # Get the permuted replicate
  md_rep <- df_perm %>%
    filter(replicate == n_run)
  # Set runs as good
  bad_run <- FALSE
  # Attempt the model
  model <- withCallingHandlers(
    expr = {
      # Build linear model on permuted data
      lmer(pct_myeloid ~ perm_area + (1 | Patient / orig.ident), data = md_rep, REML = TRUE)},
    warning = function(w) {
    # Note if the model can't converge
    bad_run <<- TRUE
    non_conv <<- non_conv + 1
    n_run <<- n_run + 1
    print("Couldn't converge")
    invokeRestart("muffleWarning")
    }
  )
      
  # Save the t val from the model if it has no errors
  if(!bad_run){
    # Get and save t val
    t_val_perm <- coef(summary(model))["perm_areaBoundary","t value"]
    t_val_perm.vec <- c(t_val_perm.vec, t_val_perm)
    # Count that we've saved a run
    n_saved <- n_saved + 1
    n_run <- n_run + 1
  } else {
    print("Skipping model")
  }
}
