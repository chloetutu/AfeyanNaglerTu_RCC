result_list <- list()

# Function to extract core name pattern
extract_core_name <- function(name) {
  matches <- regmatches(name, regexec("\\d+_(Puck\\d+_\\d+[A-Z]{2})_.*", name))
  if (length(matches[[1]]) > 1) {
    return(matches[[1]][2])  # Extracts "Puck10_102TA", "Puck5_90MB", etc.
  }
  return(NA)
}

# Loop through each CSV file
for (file in csv_files) {
  # Read CSV file
  data <- read.csv(file, stringsAsFactors = FALSE)
  
  # Extract the file name (without full path) for naming
  file_name <- data$LR[1]
  
  # Ensure the data has a column containing the relevant name
  if (!"puck" %in% colnames(data)) {
    stop("Ensure the CSV files contain the relevant column name!")
  }
  
  # Extract core names
  data$core_name <- sapply(data$puck, extract_core_name)

  # Filter out NA values (invalid rows)
  data <- data[!is.na(data$core_name), ]
  
  # Add a new column indicating the source file (to track where data came from)
  data$source_file <- file_name
  
  # Group by core name and split into sub-dataframes (4-row groups)
  grouped_data <- split(data, data$core_name)

  # Merge with existing list entries if they exist
  for (core_name in names(grouped_data)) {
    if (!is.null(result_list[[core_name]])) {
      # Append the named dataframe to the existing core name list
      result_list[[core_name]][[file_name]] <- grouped_data[[core_name]]
    } else {
      # Create a new list for the core name with a named dataframe
      result_list[[core_name]] <- list()
      result_list[[core_name]][[file_name]] <- grouped_data[[core_name]]
    }
  }
}

6 Format

new_result_list <- list()

# Function to extract and shorten the core name (removing final A/B letter)
extract_short_name <- function(core_name) {
  matches <- regmatches(core_name, regexec("Puck\\d+_(\\d+[TM])", core_name))
  if (length(matches[[1]]) > 1) {
    return(matches[[1]][2])  # Extracts "108T", "102T", etc.
  }
  return(NA)
}

for (core_name in names(result_list)) {
  short_name <- extract_short_name(core_name)  # Extract patient group name
  
  if (is.na(short_name)) next  # Skip if extraction fails

  # Initialize list to store dataframes for this patient group
  patient_dfs <- list()
  
  # Iterate over each named dataframe (corresponding to a CSV file)
  for (file_name in names(result_list[[core_name]])) {
    df <- result_list[[core_name]][[file_name]]

    # Ensure required columns exist
    if (!all(c("pval", "distance", "LR") %in% colnames(df))) {
      next  # Skip if the dataframe is missing necessary columns
    }
    
    # Create column names dynamically based on core_name and distance
    df <- df %>%
      mutate(new_column_name = paste0(core_name, "_dis", distance)) %>%
      select(LR, new_column_name, pval) %>%
      pivot_wider(names_from = new_column_name, values_from = pval)

    # Append to patient_dfs list
    patient_dfs <- append(patient_dfs, list(df))
  }
  
  # If we collected data, merge all patient dataframes **properly**
  if (length(patient_dfs) > 0) {
    patient_df <- bind_rows(patient_dfs)  # Correctly stack all dataframes

    # Ensure no duplicate rows by keeping first non-NA value per column
    patient_df <- patient_df %>%
      group_by(LR) %>%
      summarise(across(everything(), ~ dplyr::first(na.omit(.), default = NA)), .groups = "drop")

    # Store the final combined dataframe in new_result_list
    new_result_list[[core_name]] <- patient_df
  }
}

7 Browns method

library(ActivePathways)

patients <- c("102T", "107T", "108T", "109T", "110T", "111T")
distances <- c("dis15", "dis30", "dis100", "dis300")

# Browns method of combining dependent p-vals
pval_list <- list()
for(pt in patients){
  dist_list <- list()
  for(dist in distances){
    pval_mat <- perpatient.list[[pt]] %>%
      select(LR, ends_with(dist)) %>%
      column_to_rownames("LR") %>%
      replace(is.na(.), 1) %>%
      as.matrix()
  
    adj_pval <- merge_p_values(scores = pval_mat, method = "Brown")
    
    dist_list <- append(dist_list, list(adj_pval))
  }
  names(dist_list) <- distances
  pval_list <- append(pval_list, list(dist_list))
}

names(pval_list) <- patients

8 FDR correction

# Perform a Benjamini-Hochberg correction to understand which meet significance in the context of a dataset
padj_bh <- function(df){
  corrected <- p.adjust(df, method = "BH", n = length(df)) %>%
    as.data.frame() %>%
    rownames_to_column("LR") %>%
    dplyr::rename("BH" = ".") 
  return(corrected)
  }

corrected_list <- lapply(pval_list, function(inner) {
  lapply(inner, padj_bh)
})
