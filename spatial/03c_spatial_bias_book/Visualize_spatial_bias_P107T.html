14 Get pvals from the list

get_pval <- function(lis)(
  return(lis[["pval"]])
)

tls1.df <- lapply(tls1.list, get_pval) %>%
  as.data.frame() %>%
  t() 

tls2.df <- lapply(tls2.list, get_pval) %>%
  as.data.frame() %>%
  t()

14.1 Append dataframes together and correct for FDR

tls1.df <- tls1.df %>% 
  as.data.frame() %>% 
  rownames_to_column("clone_id") %>% 
  dplyr::rename("TLS1_pval" = "V1") %>%
  mutate(TLS1_pval = case_when(TLS1_pval == 0 ~ 1/5000,
                               T ~ TLS1_pval),
         TLS1_qval = p.adjust(TLS1_pval, method="BH"))
  
tls2.df <- tls2.df %>% as.data.frame() %>% 
  rownames_to_column("clone_id") %>% 
  dplyr::rename("TLS2_pval" = "V1") %>% 
  mutate(TLS2_pval = case_when(TLS2_pval == 0 ~ 1/5000,
                               T ~ TLS2_pval),
         TLS2_qval = p.adjust(TLS2_pval, method="BH"))

tls.df <- tls1.df %>%
  full_join(tls2.df)
Joining with `by = join_by(clone_id)`

14.2 Reorder data based on minimum significance

reorder.df <- tls.df %>%
  rowwise() %>%
  mutate(min_qval = min(TLS1_qval, TLS2_qval, na.rm = TRUE)) %>%
  arrange(min_qval) %>%
  ungroup()

14.3 Plot heatmap without enrichment

reorder.mat <- reorder.df %>%
  arrange(min_qval) %>%
  select(clone_id, TLS1_qval, TLS2_qval) %>%
  column_to_rownames("clone_id") %>%
  as.matrix()

col_fun = colorRamp2(c(0, 0.05, 0.06, 1), c("yellow", "white", "purple", "purple4"))
