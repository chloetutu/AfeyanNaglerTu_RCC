add_TLS_column <- function(obj){
  obj <- obj %>%
    mutate(tls_status = case_when(area %in% c("TLS1", "TLS2", "TLS3", "TLS4", "M2") ~ "inTLS",
                              area %in% c("Boundary", "Tumor"
                                          ) ~ "outTLS"))
  
  return(obj)
}

cast_shared_cd8s_102 <- function(md){
  md <- md %>%
    mutate(shared_tex_id = case_when(str_detect(sctcr_clone, "p102_295($|,)") ~ "p102_295",
                                  str_detect(sctcr_clone, "p102_289($|,)") ~ "p102_289",
                                  str_detect(sctcr_clone, "p102_737($|,)") ~ "p102_737",

# Get permuted test statistic for each gene set
model_perm <- vector("list", length(new_gene_sets))

n_perm <- 1000

# Count non-convergences
non_convergences <- vector("list", length(new_gene_sets))
names(non_convergences) <- names(new_gene_sets)

# Count times tested
tested <- vector("list", length(new_gene_sets))
names(tested) <- names(new_gene_sets)

# Loop across each gene set
for (gs in names(new_gene_sets)) {
  n_run <- 1
  n_saved <- 1
  non_conv <- 0
  
  print(gs)
  t_val_perm.vec <- c()
  # Do 1000 converging loops
  while(n_saved <= n_perm){
    # Print every saved 100 perms
    if (n_saved %% 100 == 0) { print(n_saved) }
    # Get the permuted replicate
    md_rep <- md_perm %>%
      filter(replicate == n_run)
    # Set runs as good
    bad_run <- FALSE
    # Attempt the model
    model <- withCallingHandlers(
      expr = {
        # Build formula with permutated TLs status
        f <- as.formula(paste(gs, "~ perm_tls_status + (1 | Patient/orig.ident)"))
        # Build linear model on permuted data
        lmer(f, data = md_rep, REML = TRUE)},
      warning = function(w) {
      # Note if the model can't converge
      bad_run <<- TRUE
      non_conv <<- non_conv + 1
      n_run <<- n_run + 1
      print("Couldn't converge")
      invokeRestart("muffleWarning")
      }
    )
      
    # Save the t val from the model if it has no warning messages
    if(!bad_run){
      # Get and save t val
      t_val_perm <- coef(summary(model))["perm_tls_statusinTLS","t value"]
      t_val_perm.vec <- c(t_val_perm.vec, t_val_perm)
      # Count that we've saved a run
      n_saved <- n_saved + 1
      n_run <- n_run + 1
    } else {
      print("Skipping model")
    }
  }
  model_perm[[gs]] <- t_val_perm.vec
  non_convergences[[gs]] <- non_conv
}
